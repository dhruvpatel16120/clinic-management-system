rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if the user is authenticated and has a verified email
    function isVerifiedUser() {
      return request.auth != null && request.auth.token.email_verified == true;
    }

    // Allow authenticated, verified users to read and write to appointments collection
    match /appointments/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated users to read staffData. But only the user themselves can write to their own staffData.
    // Also, allow initial creation during signup (when email is not verified yet!)
    match /staffData/{userId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == userId;
      allow update: if isVerifiedUser() && request.auth.uid == userId && !request.resource.data.diff(resource.data).affectedKeys().hasAny(['role', 'email']);
      allow delete: if false;
    }
    
    // Allow authenticated, verified users to read and write to test collection (for debugging)
    match /test/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to prescriptions collection
    match /prescriptions/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to medicines collection
    match /medicines/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to invoices collection
    match /invoices/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to payments collection
    match /payments/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to patients collection
    match /patients/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Allow authenticated, verified users to read and write to services collection
    match /services/{document} {
      allow read, write: if isVerifiedUser();
    }
    
    // Default rule - deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
