rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow authenticated users to read and write to appointments collection
    match /appointments/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to read and write to staffData collection
    match /staffData/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to read and write to test collection (for debugging)
    match /test/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Default rule - deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
